    V = lambda x: 2000 - 2000 * ( 1 - x / 40 ) ** 2
    x1 = 0
    x2 = 30
    dy = V(x2) - V(x1)
    dx = x2 - x1
    stigning = dy/dx
    print(f"Stigningstallet til den rette linjen som går mellom punktene (0, V(0)) og (30, V(30)) i funksjonen V, er {stigning:0.1f}")
  \end{pythoncode}
